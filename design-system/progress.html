Code

Progress Bar
progress-bar.html
<div class="progress-wrap">
  <div class="progress-label-row">
    <span class="progress-label">Uploading…</span>
    <span class="progress-value">65%</span>
  </div>
  <div class="progress-track progress-track--md"
       role="progressbar" aria-valuenow="65"
       aria-valuemin="0" aria-valuemax="100">
    <div class="progress-fill progress-fill--primary"
         style="width: 65%"></div>
  </div>
</div>
Progress Circle
progress-circle.html
<!-- circumference = 2π × r = 2π × 40 ≈ 251.2 -->
<!-- stroke-dashoffset = circumference × (1 - pct/100) -->
<div class="progress-circle" style="width:100px;height:100px;"
     role="progressbar" aria-valuenow="65"
     aria-valuemin="0" aria-valuemax="100">
  <svg width="100" height="100" viewBox="0 0 100 100">
    <circle class="progress-circle-track"
            cx="50" cy="50" r="40" stroke-width="8"/>
    <circle class="progress-circle-fill"
            cx="50" cy="50" r="40" stroke-width="8"
            stroke-dasharray="251.2"
            stroke-dashoffset="87.9"/>
  </svg>
  <div class="progress-circle-label">
    <span class="progress-circle-pct">65%</span>
  </div>
</div>
